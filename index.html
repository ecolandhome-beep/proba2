function analyze() { ctx.clearRect(0, 0, canvas.width, canvas.height); ctx.drawImage(video, 0, 0, canvas.width, canvas.height); const size = 200; const xS = (canvas.width - size) / 2; const yS = (canvas.height - size) / 2; // Алгоритм поиска углов квадрата const imgData = ctx.getImageData(xS, yS, size, size); const d = imgData.data; let points = []; // Простой детектор углов для поиска границ маркера for (let i = 0; i < d.length; i += 16) { const b = (d[i] + d[i+1] + d[i+2]) / 3; if (b < 50) { // Ищем черную рамку маркера const px = (i / 4) % size; const py = Math.floor((i / 4) / size); points.push({x: px, y: py}); } } if (points.length > 100) { // Вычисляем охватывающий прямоугольник (Bounding Box) const minX = Math.min(...points.map(p => p.x)); const maxX = Math.max(...points.map(p => p.x)); const minY = Math.min(...points.map(p => p.y)); const maxY = Math.max(...points.map(p => p.y)); // РИСУЕМ 3D-ОСИ (Эмуляция) ctx.strokeStyle = "#0f0"; ctx.lineWidth = 3; ctx.strokeRect(xS + minX, yS + minY, maxX - minX, maxY - minY); // Рисуем "ось Z" (показываем, что видим наклон) ctx.beginPath(); ctx.moveTo(xS + (minX + maxX)/2, yS + (minY + maxY)/2); ctx.lineTo(xS + (minX + maxX)/2, yS + minY - 20); ctx.strokeStyle = "blue"; ctx.stroke(); statusText.innerText = "МАРКЕР ЗАХВАЧЕН. СИСТЕМА ВИДИТ УГОЛ."; resVal.innerText = "READY FOR 3D"; } else { statusText.innerText = "ПОИСК МАРКЕРА..."; } }
